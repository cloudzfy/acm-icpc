#include <stdio.h>
#include <string.h>
#include <stdlib.h>
int cmp(const void *a,const void *b)
{
    return ((int *)a)[1]-((int *)a)[0]-((int *)b)[1]+((int *)b)[0];
}
int main()
{
    int f[5001],p[500][3],max,i,j,n,m;
    while(scanf("%d%d",&n,&m)!=EOF)
    {
        memset(f,-1,sizeof(f));
        f[0]=0;
        for(i=0;i<n;i++)
        {
            scanf("%d%d%d",&p[i][0],&p[i][1],&p[i][2]);
        }
        qsort(p,n,sizeof(p[0]),cmp);
        for(i=0;i<n;i++)
        {
            for(j=m-p[i][0];j>0;j--)
            {
                if(f[j]!=-1&&j+p[i][0]>=p[i][1]&&(f[p[i][0]+j]<f[j]+p[i][2]))
                    f[p[i][0]+j]=f[j]+p[i][2];
                if(f[j]!=-1&&j+p[i][0]<p[i][1]&&(f[p[i][1]]<f[j]+p[i][2]))
                    f[p[i][1]]=f[j]+p[i][2];
            }
            if(f[p[i][1]]==-1||f[p[i][1]]<p[i][2])
                f[p[i][1]]=p[i][2];
        }
        max=0;
        for(i=1;i<=m;i++)
        {
            if(max<f[i])max=f[i];
        }
        printf("%d\n",max);
    }
    return 0;
}
